  160.         // open the cache file for writing

  162.         // save the contents of the output buffer to the file
  163.         fwrite($file, ob_get_contents());        
  164.         //turn off output buffer
  165.         ob_end_clean();
  166.         // close the file
  167.         fclose($file);
  168.     }

  202.             //if secure, add secure stylesheets - all in style/secure/
  203.             if(Security::clearance()) { foreach(new DirectoryIterator('style/secure') as $style) { 
  204.                 //make sure file is .css or .css.php
  205.                 if( preg_match("/\.css$/",$style) || preg_match("/\.css\.php$/",$style))
  206.                     $this->addStyle('style/secure',$style,'secure'); }}
  207.             
  208.             //set all styles for main template
  209.             $this->main->styles = array_merge($this->stylesPublic, $this->stylesSecure);
  210.         }
  211.         if($scripts) {
  212.             //LOAD codeCore JavaScripts
  213.                 //Grab all public JavaScripts - all in codeCore/js/
  214.                 foreach(new DirectoryIterator('codeCore/js') as $script) { 
  215.                 //do not include debug.js here - that is handled in the constructor
  216.                 if($script != 'debug.js')

  218.                     if(preg_match("/\.js$/",$script))
  219.                         $this->addScript('codeCore/js',$script); 
  220.                 }
  221.             //LOAD codeSite JavaScripts
  222.                 //Grab all public JavaScripts - all in codeSite/js/
  223.                 foreach(new DirectoryIterator('codeSite/js') as $script) { 
  224.                     //make sure file is .js
  225.                     if(preg_match("/\.js$/",$script))
  226.                         $this->addScript('codeSite/js',$script); 
  227.                 }
  228.             //set all scripts for main template
  229.             $this->main->scripts = array_merge($this->scriptsPublic, $this->scriptsSecure);

  231.
