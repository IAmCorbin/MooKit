  136. /** check an array for existing keys, checks each key with array_key_exists
  137.   * @param     array     $keyArray    an array containing all the keys to check for
  138.   * @param     array     $array         the array to test - passed by reference
  139.   * @param     bool     $setBlank    switch to set missing key values to empty strings instead of returning false
  140.   * @param     bool     $blankChk    switch to check for blank values and return false if found
  141.   * @return    bool        true/false

  143.  function array_keys_exist($keyArray, &$array, $setBlank=FALSE, $blankChk=FALSE) {
  144.     foreach($keyArray as $key) {
  145.         if(!array_key_exists($key, $array)) {
  146.             if($setBlank) {
  147.                 $array[$key] = '';
  148.             } else
  149.                 return false;
  150.         }
  151.         if($blankChk)
  152.             if($array[$key] == '')
  153.                 return false;
  154.     }
  155.     return true;
  156. }

  177. /**
  178.   * print a line break with an optional centered message - used for debugging
  179.   * @param    string    $msg    the message
  180.   * @param    int        $rows    the number of rows
  181.   * @param    string    $char    the character to use

  184. function linebreak($msg=NULL,$rows=1,$char="~",$cols=100) {
  185.     if($msg && !$rows) $rows = 4;
  186.     for($x = 0; $x < $rows; $x++) {
  187.         for($y = 0; $y < $cols; $y++)
  188.             echo $char;
  189.         echo "<br />";
  190.         //optional message
  191.         if($msg && $x==round($rows/2)-1) {
  192.             $msgLength = strlen($msg);
  193.             $msgMiddle = round(($cols/2)-($msgLength/2));
  194.             for($y = 0; $y < $cols-$msgLength; $y++) {
  195.                 echo $char;
  196.                 //echo message in the middle
  197.                 if($y == $msgMiddle) echo $msg;
  198.             }
  199.             echo "<br />";
  200.         }
  201.     }

  208. function ErrorHandler($errno, $errmsg, $filename, $linenum, $vars) {
  209.     //Error Log filename
  210.       $errorLogPath = ERROR_LOG_DIR."PHPerrors.xml";
  211.     //how long to keep errors 
  212.       defined('PHP_ERROR_EXPIRE')? $errorExpireTime = strtotime(PHP_ERROR_EXPIRE) : $errorExpireTime = NULL;
  213.     // define an assoc array of error string
  214.     // in reality the only entries we should
  215.     // consider are E_WARNING, E_NOTICE, E_USER_ERROR,
  216.     // E_USER_WARNING and E_USER_NOTICE
  217.       $errortype = array (
  218.           E_ERROR              => 'Error',
  219.           E_WARNING            => 'Warning',
  220.           E_PARSE              => 'Parsing Error',
  221.           E_NOTICE             => 'Notice',
  222.           E_CORE_ERROR         => 'Core Error',
  223.           E_CORE_WARNING       => 'Core Warning',
  224.           E_COMPILE_ERROR      => 'Compile Error',
  225.           E_COMPILE_WARNING    => 'Compile Warning',
  226.           E_USER_ERROR         => 'User Error',
  227.           E_USER_WARNING       => 'User Warning',
  228.           E_USER_NOTICE        => 'User Notice',
  229.           E_STRICT             => 'Runtime Notice',
  230.           E_RECOVERABLE_ERROR  => 'Catchable Fatal Error'
  231.           );
  232.     // set of errors for which a var trace will be saved
  233.       $user_errors = array(E_USER_ERROR, E_USER_WARNING, E_USER_NOTICE);
  234.     
  235.     if(is_readable($errorLogPath)) { 
  236.         // Create a new DOMDocument object with formatting
  237.         $errorLog = new DOMDocument('1.0');
  238.         $errorLog->formatOutput = true;
  239.         $errorLog->preserveWhiteSpace = false;
  240.         //Import error log file
  241.         $errorLog->load($errorLogPath);
  242.         //grab all errors
  243.         $errors = $errorLog->documentElement->getElementsByTagName("error");
  244.         
  245.         
  246.         //check time and remove old errors
  247.         if($errorExpireTime) {
  248.             $errorsLength = $errors->length-1;
  249.             for($n = $errorsLength; $n >= 0; $n--) {
  250.                 $error = $errors->item($n);
  251.                 //get error time
  252.                 $errorTime = $error->getElementsByTagName("datetime");
  253.                 //convert time
  254.                 $errorTime = strtotime($errorTime->item(0)->nodeValue);
  255.                 //compare and remove if expired
  256.                 if( $errorTime < $errorExpireTime)
  257.                     $errorLog->documentElement->removeChild($error);
  258.             }
  259.         }
  260.         
  261.         //Create the new Error XML
  262.         $err = new SimpleXMLElement("<error></error>");
  263.         $err->addChild('datetime',date('d M Y H:i T'));
  264.         $err->addChild('num',$errno);
  265.         $err->addChild('type',$errortype[$errno]);
  266.         $err->addChild('msg',$errmsg);
  267.         $err->addChild('scriptname',$filename);
  268.         $err->addChild('linenum',$linenum);
  269.     
  270.         if (in_array($errno, $user_errors)) {
  271.             ob_start();
  272.             var_export($vars);
  273.             $vars = ob_get_contents();
  274.             ob_clean(); //cleanup ob
  275.             $err->addChild('vartrace', $vars);
  276.         }
  277.         //convert the simpleXML to a DOMElement 
  278.         $Err = dom_import_simplexml($err);
  279.         //import the node, and all its children, to the document
  280.         $Err = $errorLog->importNode($Err,true);
  281.         // And then append it to the "<root>" node
  282.         $errorLog->documentElement->appendChild($Err);
  283.         
  284.         // save the modified error log
  285.         $errorLog->save($errorLogPath);
  286.     } else { 
  287.         // SEND AN EMAIL TO ADMINISTRATOR IF ERROR LOGGING IS BROKEN
  288.     }
  289.     
  290.     // !!!
  291.     // Add Administrator Email notification depending on error type
  292.     // !!!
  293.     //if ($errno == E_USER_ERROR) {
  294.     //mail("phpdev@example.com", "Critical User Error", $err);
  295.     //}
  296. }
  297. set_error_handler("ErrorHandler");
