  46.     public function extract($vars) {
  47.         if($vars)
  48.             foreach ($vars as $property => $value)
  49.                 $this->vars[$property] = $value;
  50.     }

  60.     public function __set($name, $value) {

  66.     public function __get($name) {

  68.     }
  69.     /**

  73.     public function mergevars()    {

  75.             return array_merge($this->parent->mergevars(), $this->vars);
  76.         else
  77.             return $this->vars;
  78.     }

  88.         ob_start();

  96.         ob_end_clean();

  101.                 $this->result = gzencode($this->result);
  102.                 header('Content-Encoding: gzip');
  103.             }
  104.         }
  105.         //return the results of the merge for this template
