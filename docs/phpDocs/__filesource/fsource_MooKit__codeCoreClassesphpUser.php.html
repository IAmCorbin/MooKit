  342.     /**

  346.      *@return     string     encrypted password

  348.     public function encryptPassword($pass,$regTime) { 
  349.         //create salt from the sha1 of the user's registration time
  350.         $salt = sha1($regTime);
  351.         //md5 encrypt the salt+password, then sha1 that whole thing and return the encrypted password
  352.         return sha1(md5($salt.$pass));
