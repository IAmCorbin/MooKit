1 /**
  2  * @class A Class to enable the display of assigned sub elements...Link -> sublinks
  3  * @author Corbin Tarrant
  4  *  ___            ___         __ 
  5  *    |   /\  |\/| |       __   |_/  |__   -   _
  6  *  _|_ /  \ |  | |___  |__|  | \  |__|  |  |  |
  7  * 
  8  * {@link http://www.IAmCorbin.net }
  9  * @version 1
 10  * @birth April 30th, 2010
 11  * @package MooKit
 12  * 
 13  * @requires MooTools 1.2
 14  * {@link http://mootools.net/}
 15  *
 16  * @property 	{element}	linkContainers		all of the containers holding the links that contain sublinks
 17  * @property 	{element}	sublinkContainers		all of the sublink elements
 18  */
 19 var SubDisplay = new Class({
 20 	Implements: [Options,Events],
 21 	options: {
 22 		sublinkClass: 'sublink',
 23 		fadeDelay: 1500
 24 	/** 
 25 	  * @constructor
 26 	  * @param 	{element}[]	sublinks	all of the sublink containers
 27 	  */
 28 	},initialize: function() {
 29 		//grab elements
 30 		this.sublinkContainers =$$("."+this.options.sublinkClass);
 31 		this.linkContainers = this.sublinkContainers.getParent('.link');
 32 		this.links = this.linkContainers.getFirst('a');
 33 		this.sublinks = this.sublinkContainers.getFirst('a');
 34 		//Initialize Positioning and display for each set of sublinks
 35 		this.sublinkContainers.setStyle('display','none');
 36 		this.linkContainers.each(function(link) {
 37 			LOC = getXY(link);
 38 			var sublinks = link.getElements('.sublink');
 39 			sublinks.setStyles({
 40 				top: LOC.Y+link.getStyle('lineHeight').toInt()+10+'px',
 41 				left: LOC.X
 42 			});
 43 			//offset sublinks
 44 			var offsetSubs = 0;
 45 			sublinks.each(function(sublink) {
 46 				//setup FX chaining
 47 				sublink.set('morph',{ link:'chain'});
 48 				if(offsetSubs==0) { //don't offset first sublink, just set initial positioning
 49 					offsetSubs = sublink.getStyle('left').toInt();
 50 				} else {
 51 					previousSize = sublink.getPrevious().getElement('a').get('text');
 52 					offsetSubs +=  previousSize.length*7;
 53 					sublink.setStyle('left', offsetSubs+'px');
 54 				}
 55 			});
 56 		});
 57 		this.sublinkContainers.tween('opacity',0);
 58 		//ADD LINK EVENTS
 59 		this.links.addEvents({
 60 			//Mouse Enter Display
 61 			mouseenter: function(e) { 
 62 				var sublink = e.target.getParent().getLast();
 63 				if(!sublink.retrieve('display')) {
 64 					sublink.store('display',1);
 65 					this.displaySublinks(e.target.getParent()); 
 66 				}
 67 			}.bind(this), 
 68 			//Mouse Move Display if Hidden
 69 			mousemove: function(e) {
 70 				sublink = e.target.getParent().getLast();
 71 				if(!sublink.retrieve('mmDisplay')) {
 72 					sublink.store('mmDisplay',1);
 73 					(function() { 
 74 						if(sublink.getStyle('display') == 'none') {
 75 							this.displaySublinks(e.target.getParent());
 76 						}
 77 					}.bind(this)).delay(this.options.fadeDelay);
 78 				}
 79 				
 80 			}.bind(this),
 81 			//vanish
 82 			mouseleave: function(e) { 
 83 				var sublinks = e.target.getParent().getElements('.sublink');
 84 				this.delayedVanish(sublinks);
 85 			}.bind(this)
 86 		});
 87 		//ADD SUBLINK EVENTS
 88 		this.sublinks.addEvents({
 89 			mouseenter: function() {
 90 				//cancel vanishing
 91 				var sublinkDestructor = this.getParent('span[class="link"]').getElements('span[class="sublink"]').getLast().retrieve('destructorID');
 92 				$clear(sublinkDestructor);
 93 				sublink.eliminate('destructorID');
 94 			},
 95 			mouseleave: function(e) {
 96 				var sublinks = e.target.getParent().getParent().getElements('.sublink');
 97 				this.delayedVanish(sublinks);
 98 			}.bind(this)
 99 		});
100 	},
101 	/** 
102 	  * display the sublinks of passed in link
103 	  * @param 	{element} 	link		the link you want to display sublinks for
104 	  */
105 	displaySublinks: function(link) {
106 		var sublinks = link.getElements('.sublink');
107 		//cancel vanishing - clear any anonymous vanishing functions that may be about to trigger hideSublinks
108 		var sublinkDestructor = sublinks.getLast().retrieve('destructorID');
109 		$clear(sublinkDestructor);
110 		sublinks.getLast().eliminate('destructorID');
111 		//display sublinks
112 		sublinks.each(function(sublink) {
113 			sublink.morph({display: 'block'});
114 			sublink.morph({opacity: '1'});
115 		});
116 	},
117 	delayedVanish: function(sublinks) {
118 		//setup to fade links
119 		var sublinkDestructor = (function() { 
120 			this.hideSublinks(sublinks);
121 		}.bind(this)).delay(this.options.fadeDelay);
122 		//store id so we can cancel if user's mouse reenters a sublink or parent link
123 		sublinks.getLast().store('destructorID',sublinkDestructor);
124 	},
125 	/**
126 	  * hide the passed in sublink elements
127 	  * @var 	{element}[]	sublinks		the sublinks to hide
128 	  */
129 	hideSublinks: function(sublinks) {
130 		sublinks.each(function(sublink) {
131 			sublink.morph({opacity: '0'});
132 			sublink.morph({display: 'none'});
133 		});
134 		//eliminate all flags so we can retrigger sublink display
135 		sublinks.getLast().eliminate('destructorID');
136 		sublinks.getLast().eliminate('mmDisplay');
137 		sublinks.getLast().eliminate('displayID');
138 	}
139 });
140 
141 window.addEvent('domready', function() {
142 
143 	//Setup Ajax Links	
144 	var links = $$('.ajaxLink');
145 	links.each(function(link) {
146 		link.addEvent('click',function(e) {
147 			e.stop();
148 			window.location.hash = link.get('href');
149 		});
150 	});
151 
152 	sublinks = new SubDisplay();
153 
154 	//Description Boxes
155 	var descs = $$('.linkDesc');
156 	var descFadeSpeed = 1500;
157 	descs.set('tween', { duration: descFadeSpeed });
158 	descs.setStyle('opacity','0');
159 	//add event to each description's cooresponding link
160 	descs.getPrevious().addEvents({
161 		mouseenter: function() {
162 			var desc = this.getNext();
163 			desc.set('tween', { duration: descFadeSpeed });
164 			desc.fade(0.75);
165 		},
166 		mousemove: function(e) {
167 			if(e.target.getParent().hasClass('sublink')) {
168 				var X = (e.page.x)-(e.page.x)+30+'px';
169 				var Y =  (e.page.y)-(e.page.y)+30+'px';
170 			} else {
171 				var X = e.page.x+10+'px';
172 				var Y = e.page.y-10+'px';
173 			}
174 			var desc = this.getNext();
175 			desc.setStyles({
176 				left: X,
177 				top: Y
178 			});
179 		},
180 		mouseleave: function() {
181 			var desc = this.getNext();
182 			desc.set('tween', { duration: descFadeSpeed/3 });
183 			desc.fade(0);
184 		}
185 	});
186 	
187 	
188 });
