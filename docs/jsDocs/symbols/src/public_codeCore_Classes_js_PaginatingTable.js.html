1 /**
  2  * @class
  3  * @author {@link http://madhatted.com/2008/1/11/the-joy-of-a-minimal-complete-javascript-table-sort}
  4  * added this documentation May 13, 2010
  5  * @package MooKit
  6  * @requires MooTools 1.2 {@link http://mootools.net/}
  7  * // new PaginatingTable( 'my_table', 'ul_for_paginating', {
  8  * // per_page: 10, // How many rows per page?
  9  * // current_page: 1, // What page to start on when initialized
 10  * // offset_el: false, // What dom element to stick the offset in
 11  * // cutoff_el: false, // What dom element to stick the cutoff in
 12  * // details: false // Do we have hidden/collapsable rows?
 13  * // });
 14  * //
 15  * // The above were the defaults. You could also pass an array of paginators
 16  * // instead of just one.
 17  * //
 18  * // Requires mootools Class, Array, Function, Element, Element.Selectors,
 19  * // Element.Event, and you should probably get Window.DomReady if you're smart.
 20  * //
 21  * @property 	{element}	table 			the table to add pagination to 
 22  * @property	{element}	tbody			the body of the table
 23  * @property	{array}		paginators		paginator elements
 24  * @property	{int}		pages			the number of pages
 25  * @property	{int}		current_page		the current page number
 26  * @property	{int}		low_limit			low page limit
 27  * @property	{int}		high_limit		high page limit
 28  * @property 	{element}	options.offset_el 	offset element
 29  * @property 	{element} 	options.cutoff_el 	cutoff element
 30  * @property 	{bool} 		options.details		details switch
 31  * @property 	{int} 		options.per_page 	number of rows per page
 32  */
 33 var PaginatingTable = new Class({
 34   Implements: Options,
 35   options: {
 36     per_page: 10,
 37     current_page: 1,
 38     offset_el: false,
 39     cutoff_el: false,
 40     details: false
 41   },
 42 /** 
 43   * @constructor
 44   * @param 	{element}	table		the table to add pagination to 
 45   * @param 	{element}	ids			ids of paginators
 46   * @param	{Array}		options		set options
 47   */
 48   initialize: function( table, ids, options ) {
 49     this.table = $(table);
 50     this.setOptions(options);
 51     
 52     this.tbody = this.table.getElement('tbody');
 53     
 54     if (this.options.offset_el)
 55       this.options.offset_el = $(this.options.offset_el);
 56     if (this.options.cutoff_el)
 57       this.options.cutoff_el = $(this.options.cutoff_el);
 58 
 59     this.paginators = ($type(ids) == 'array') ? ids.map($) : [$(ids)];
 60 
 61     if (this.options.details) {
 62       this.options.per_page = this.options.per_page * 2
 63     }
 64 
 65     this.update_pages();
 66   },
 67 /** @function update pages **/
 68   update_pages: function(){
 69     this.pages = Math.ceil( this.tbody.getChildren().length / this.options.per_page );
 70     this.create_pagination();
 71     this.to_page( 1 );
 72   },
 73 /** @function change pages **/
 74   to_page: function( page_num ) {
 75     page_num = page_num.toInt();
 76     if (page_num > this.pages || page_num < 1) return;
 77     this.current_page = page_num;
 78     this.low_limit = this.options.per_page * ( this.current_page - 1 );
 79     this.high_limit = this.options.per_page * this.current_page;
 80     var trs = this.tbody.getChildren();
 81     if (trs.length < this.high_limit) this.high_limit = trs.length;
 82     for (var i = 0, j = trs.length; i < j; i++) {
 83       trs[i].style.display = (this.low_limit <= i && this.high_limit > i) ? '' : 'none';
 84     }
 85     this.paginators.each(function(paginator){
 86       var as = paginator.getElements('a').removeClass('currentPage');
 87       as[this.current_page].addClass('currentPage');
 88     }, this);
 89     if (this.options.offset_el)
 90       this.options.offset_el.set('text', Math.ceil( this.low_limit / ( this.options.details ? 2 : 1 ) + 1 ) );
 91     if (this.options.cutoff_el)
 92       this.options.cutoff_el.set('text', ( this.high_limit / ( this.options.details ? 2 : 1 ) ) );
 93   },
 94 /** @function go to next page **/
 95   to_next_page: function() {
 96     this.to_page( this.current_page + 1 );
 97   },
 98 /** @function go to previous page **/
 99   to_prev_page: function() {
100     this.to_page( this.current_page - 1 );
101   },
102 /** @function create pagination **/
103   create_pagination: function() {
104     this.paginators.each(function(paginator){
105       paginator.empty();
106       this.create_pagination_node( '«', function(evt){
107         var evt = new Event( evt );
108         this.to_prev_page();
109         evt.stop();
110         return false;
111       }).injectInside( paginator );
112       for (var page=1; page <= this.pages; page++){
113         this.create_pagination_node( page, function(evt){
114           var evt = new Event( evt );
115           this.to_page( evt.target.get( 'text' ) );
116           evt.stop();
117           return false;
118         }).injectInside( paginator );
119       }
120       this.create_pagination_node( '»', function(evt){
121         var evt = new Event( evt );
122         this.to_next_page();
123         evt.stop();
124         return false;
125       }).injectInside( paginator );
126     }.bind( this ));
127   },
128 /** @function create paginatoion node **/
129   create_pagination_node: function( text, evt ) {
130     var span = new Element( 'span' ).set( 'html', text );
131     if (text == '«'){
132       var a = new Element( 'a', { 'href': '#', 'class': 'previous-page' }).addEvent( 'click', evt.bind( this ) );
133     } else if (text == '»'){
134       var a = new Element( 'a', { 'href': '#', 'class': 'next-page' }).addEvent( 'click', evt.bind( this ) );
135     } else {
136       var a = new Element( 'a', { 'href': '#'}).addEvent( 'click', evt.bind( this ) );
137     }
138     var li = new Element( 'li' );
139     span.injectInside( a.injectInside( li ) );
140     return li;
141   }
142 
143 });
