  3  * @author Corbin Tarrant

  6  *  _|_ /  \ |  | |___  |__|  | \  |__|  |  |  |
  7  *

 11  * @package MooKit
 12  * 
 13  * @requires MooTools 1.2
 14  * {@link http://mootools.net/}
 15  *
 16  * @property 	{element}	container 			The deep linking container
 17  * @property	{Hash[]}		cache				Hash array of cached page content
 18  * @property	{int}		hashMonitor			The periodical ID of the hashMonitor ( to use for clearing later )
 19  * @property	{string}		lastHash				the last hash checked
 20  * @property 	{int} 		options.time 			Duration between hash checks (in ms)
 21  * @property 	{string[]} 	options.cache 			Preload the cache with hash objects containing page data
 22  * @property 	{bool} 		options.cookies 		Switch cookies on/off
 23  * @property 	{int} 		options.cookieLife 		Cookie life in days
 24 * @property 	{bool} 		options.DEBUG 		Switch debug messages on/off
 25 * @property 	{$empty}	options.onUpdate 		Event fires when hash has been changed and is not cached, user should pass in a function that will execute container.load with different content for different hashes
 26  */
 27 var DeepLinker = new Class({
 28 	Implements: [Options,Events],
 29 	options: {
 30 		time: 1000,
 31 		cache: [],
 32 		cookies: true,
 33 		cookieLife: 0,
 34 		DEBUG: false
 35 		/*
 36 		onUpdate: $empty
 37 		*/
 38 	/** 
 39 	  * @constructor
 40 	  */
 41 	},initialize: function(container, options) {
 42 		//set deep linking container
 43 		this.container = $(container);
 44 		//merge in passed options
 45 		this.setOptions(options);
 46 		this.cache = new Array();
 47 		this.lastCache = null;
 48 		//start monitoring hash
 49 		this.hashMonitor = (function() { this.checkHash(); }.bind(this)).periodical(this.options.time);
 50 	/** 
 51 	  * @function check the current hash 
 52 	  */
 53 	},checkHash : function() { 
 54 		if(window.location.hash !== this.lastHash) {
 55 			this.debug("|------  HASH CHANGED  :"+window.location.hash+":------|");
 56 			if(window.location.hash == '') window.location.hash = "#front"
 57 			//if(this.lastHash) {
 58 				var cacheLocation = null;
 59 				//check each page in cache to see if the data is already cached OR check for cookie
 60 				if(this.options.cookies)
 61 					var cached = Cookie.read(window.location.hash)
 62 				else
 63 					var cached = this.cache.some(function(item, index) { 
 64 										if(item.hasValue(window.location.hash)) {
 65 											cacheLocation = index;
 66 											return true;
 67 				/*if cache was found*/		}});
 68 				if(cached) {
 69 					//get cached content
 70 					if(this.options.cookies) {
 71 						this.debug("ALREADY CACHED: LOADING FROM COOKIE");
 72 						var content = Cookie.read(window.location.hash);
 73 						document.title = Cookie.read(window.location.hash+"title");
 74 					} else {
 75 						this.debug("ALREADY CACHED: LOADING FROM HASH");
 76 						var content = this.cache[cacheLocation].get('content');
 77 						document.title = this.cache[cacheLocation].get('title');
 78 					}
 79 					this.debug("content to load: "+content);
 80 					//display cached content
 81 					this.container.set('html',content);
 82 				} else {
 83 					//setup to cache when load is complete
 84 					this.container.set('load',{
 85 						onComplete: function() {
 86 							this.saveCache();
 87 						}.bind(this)
 88 					});
 89 					this.debug("LOADING CONTENT NOW - onUpdate fires");
 90 					this.fireEvent('update');
 91 				}
 92 		} else if(window.location.hash == "#kill") {
 93 			$clear(this.hashMonitor);
 94 			console.dir(this.cache);
 95 			this.debug('DeepLinker : Hash Monitor Has Been Terminated');
 96 		} else if(window.location.hash == this.lastHash)
 97 			this.debug("|------  No Hash Change   ||Cached Pages: "+this.cache.length+"|| ------|");
 98 		
 99 		//set the current hash to last hash for next check
100 		this.lastHash =  window.location.hash;
101 	/** 
102 	  * @function debugging function
103 	  */
104 	},debug: function(input) { if(this.options.DEBUG && window.console) console.log(input); 
105 	/** 
106 	  * @function save the current content to appropriate cache
107 	  */
108 	},saveCache: function() { 
109 		//do not cache secure content
110 		if(!$('LOGGEDIN')) {
111 			//cache content
112 			if(!this.lastHash) {
113 				if(this.options.cookies) {
114 					this.debug("CACHING TO COOKIE NOW - FIRST CACHE");
115 					Cookie.write(window.location.hash,this.container.get('html'), { duration: this.options.cookieLife }) ;
116 					Cookie.write(window.location.hash+"title",document.title, { duration: this.options.cookieLife }) ;
117 				} else {
118 					this.debug("CACHING TO HASH NOW - FIRST CACHE");
119 					this.cache[0] = new Hash({hash:window.location.hash,title:document.title,content:this.container.get('html')});
120 				}
121 			} else {
122 				if(this.options.cookies) {
123 					this.debug("CACHING TO COOKIE NOW : "+this.cache.length+this.container.get('html'));
124 					Cookie.write(window.location.hash,this.container.get('html'), { duration: this.options.cookieLife }) ;
125 				} else {
126 					this.debug("CACHING TO HASH NOW : "+this.cache.length+this.container.get('html'));
127 					this.cache[this.cache.length] = new Hash({hash:window.location.hash,title:document.title,content:this.container.get('html')});
128 				}
129 			}
130 		}
131 	}
132 });
