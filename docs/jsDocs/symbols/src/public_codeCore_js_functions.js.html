  8 /**
  9   * @function take a php response, check for errors, and return the JSON
 10   * @param 	{string}		response		the php response from an ajax call
 11   * @param 	{element}	errorBox 		optional id of element to display errors in
 12   */
 13 function handleResponse(response, errorBox) {
 14 	if(DEBUG)
 15 		$('debugBox').set('html',response);
 16 	//allow a reponse of 1
 17 	if(response == 1)
 18 		return response;
 19 	//make sure proper JSON was returned and not a php error
 20 	if(!response.test('^\\[?(\{[^\}]+\}\,?)+\\]?$')) {
 21 		if(DEBUG) 
 22 			alert('JSON ERROR! : '+response);
 23 		
 24 		if(errorBox) {	
 25 			//display an error message to the user
 26 			$(errorBox).setStyle('display','block');
 27 			$(errorBox).set('html',"Fatal Error. Please Contact Administrator if this persists");
 28 		}
 29 		
 30 		//Log Error
 31 		new Request.JSON({
 32 			url: 'codeCore/php/logError.php',
 33 			onSuccess: function(responseJSON, responseTEXT) {
 34 				debug("PHP error logged");
 35 			}
 36 		}).send('json={"error": "'+response.clean()+'"}');
 37 		
 38 		return false;
 39 	}
 40 	//decode JSON and return
 41 	return JSON.decode(response);
 42 }

 44 /**
 45   * @function take a php response, check for errors, and return the JSON
 46   * @param 	{string}		status	json.status
 47   * @param 	{element}	errorBox 	optional id of element to display errors in
 48   */
 49 function checkJSONerrors(status, errorBox) {
 50 	switch(status) {
 51 		case "E_MISSING_DATA":
 52 			$(errorBox).set('html',"Missing Data");
 53 			break;
 54 		case "E_FILTERS":
 55 			$(errorBox).set('html',"Invalid Input, please try again or contact the administrator if this problem persists");
 56 			break;
 57 		case "E_BADPASS":									
 58 			$(errorBox).set('html',"The passwords you entered do not match");
 59 			break;
 60 		case "E_DUPLICATE":
 61 			$(errorBox).set('html',"Alias or Email Address already exists, try a different alias or email address");
 62 			break;
 63 		case "E_INSERT":
 64 			$(errorBox).set('html',"Error adding user, please try again later. If problem persists contact the administrator");
 65 			break;
 66 		case "E_ID":
 67 			$(errorBox).set('html',"Invalid input, please try again or contact the administrator if this problem persists");
 68 			break;
 69 		case "E_NOAUTH":
 70 			$(errorBox).set('html',"Unauthorized");
 71 	}
 72 }

 74 /**
 75   * @function dynamically add array of CSS Stylesheets and JavaScripts
 76   * @param 	{Array}	styles 	array of styles ( full directory path and filenames )
 77   * @param 	{Array}	scripts 	array of scripts ( full directory path and filenames )
 78   */
 79 function addAssets(styles,scripts) {
 80 	//if previously loaded, destroy old css and javascript 
 81 	styles.each(function(style) { 
 82 		style = 'CSS'+style.replace(/.+\//,"").replace(/\.css\.php$/,"").replace(/\.css$/,"");
 83 		if($(style))	$(style).destroy();
 84 	});
 85 	scripts.each(function(script) {
 86 		script = 'JS'+script.replace(/.+\//,"").replace(/\.js$/,"");
 87 		if($(script))	$(script).destroy();
 88 	});
 89 						
 90 	//load CSS
 91 	styles.each(function(style) { //for id : strip directory and extention, prepend "CSS"
 92 		new Asset.css(style, { id: 'CSS'+style.replace(/.+\//,"").replace(/\.css\.php$/,"").replace(/\.css$/,"") });
 93 	});
 94 	//load javascript
 95 	scripts.each(function(script) { //for id : strip directory and extention, prepend "JS"
 96 		new Asset.javascript(script, { id: 'JS'+script.replace(/.+\//,"").replace(/\.js$/,"") });
 97 	});
 98 }

119 /**

121   * @param 	{string}	module		the module name
122   * @param 	{string}	container		the id of the container to load into
123   */
124 function CORE_LOAD(module, container) {
125 	switch(module) {

133 			});
134 			fancyLoad($(container),'codeCore/php/secure/adminPanel.php');
135 			break;
136 		case 'CreatePanel':
137 			$(container).set('load',{
138 				onSuccess: function(r1,r2,r3) {

146 		default:
147 			break;
148 	}

151 /**
152   * @function create smooth ajax loading (fade out/in)
153   * @param 	{element}	container		the container to load into (also the fading element)
154   * @param 	{string}		url			what to load
155   * @param 	{int}		speed		fade speed
156   */
157 function fancyLoad(container, url, speed) {
158 	speed = typeof(speed) == "undefined" ? 250 : speed;
159 	container.fade(0);
160 	( function() {container.load(url); } ).delay(speed);
161 	( function() { container.fade(1); } ).delay(speed*2);
162 }

164 /**
165   * Find the absolute position of an element on the page by iterating through all parents
166   * @param	{element}	obj		the object to get position for
167   * @returns 	an object with the X and Y position
168   */
169 function getXY(obj) {
170 	if (obj) {
171 		var curleft = 0;
172 		var curtop = 0;
173 		if (obj.offsetParent) {
174 			curleft = obj.offsetLeft;
175 			curtop = obj.offsetTop;
176 			while (obj = obj.offsetParent) {
177 				curleft += obj.offsetLeft;
178 				curtop += obj.offsetTop;
179 			}
180 		}
181 		return { X: curleft, Y: curtop };
182 	}
