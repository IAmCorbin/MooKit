1 /**
  2  * @class The Lightbox Class creates a layer that can be toggled on and off and contain multiple elements of content
  3  * @birth Febuary 18th, 2010
  4  * 
  5  * To use:
  6  *   1. Create a div on the page with an ID of your choice
  7  *   2. Create as many content divs as you need with the class name <ID of div created in step 1>Content
  8  *   3. Create another element on the page with ID of <ID of div created in step 1>Close
  9  *		Ex:
 10  *			<div id="lightbox"></div>
 11  *			<div class="lightboxContent">Place Content Here</div>
 12  *			<span class="lightboxClose">Close Lightbox</div>
 13  *   4. In your mootools domready event create a new lightbox passing in the ID of div from step 1 and any of the options desired
 14  *		Ex:
 15  *			lightbox = new Lightbox('lightbox');
 16  *   5. Create a way to set off the initial trigger
 17  *		Ex:
 18  *			html : <div id="open"></div>
 19  *			js: $('open').addEvent('click',function() { lightbox.trigger });
 20  *   6. Enjoy!
 21  *
 22  *			~~~~~~~~~~~~~~~
 23  *			~~Free for all to use~~
 24  *			~~~~~~~~~~~~~~~
 25  *
 26  * @author Corbin Tarrant
 27 *   ___            ___         __ 
 28  *    |   /\  |\/| |       __   |_/  |__   -   _
 29  *  _|_ /  \ |  | |___  |__|  | \  |__|  |  |  |
 30  * 
 31  * @link http://www.IAmCorbin.net
 32  * @version 1.0
 33  *
 34  * @requires MooTools 1.2
 35  * {@link http://mootools.net/}
 36  * 
 37  * @package MooKit
 38  *
 39  * @property	{Element}  	lightbox 				the lightbox element
 40  * @property	{bool}	 	hidden				visablility flag
 41  * @property	{Element}	closeButton		 	the element that closes the lightbox
 42  * @property	{Element[]}	content 				array of all the content boxes in the lightbox
 43  *
 44  * @property	{bool}		options.fade 			Should the lightbox fade or just snap in and out
 45  * @property	{int}		options.fadeSpeed 		(if fade = true) how long should the lightbox take to fade in and out
 46  * @property	{int}		options.removeDelay 	Delay before lightbox is removed from display - time starts after fade is complete
 47  * @property	{$empty}	options.onShow		Fires immediately after lightbox is added to display and fades in
 48  * @property	{$empty}	options.onHide			Fires immediately after a trigger() call when lightbox is already displayed
 49  * @property	{$empty}	options.onRemove		Fires when lightbox has been removed from display
 50  */
 51 var LightBox = new Class({
 52 	Implements: [Options,Events],
 53 	options: {
 54 		fade: true,
 55 		fadeSpeed: '200',
 56 		removeDelay: '0'
 57 		/*
 58 		onShow: $empty,
 59 		onHide: $empty.
 60 		onRemove: $empty
 61 		*/
 62 	},
 63 	/**
 64 	  * @constructor
 65 	  * @param {string} 	lightbox 		the ID of the lightbox
 66 	  * @param {array} 	options 		passed in options
 67 	  */
 68 	initialize: function(lightbox, options) {
 69 		this.setOptions(options);
 70 		//make sure the delay for removing the lightbox takes at least as long as the fade
 71 		this.options.removeDelay = this.options.removeDelay.toInt()+this.options.fadeSpeed.toInt();
 72 		//visability flag - hidden at start
 73 		this.hidden = true;
 74 		//get lightbox and make transparent
 75 		this.lightbox = $(lightbox);
 76 		this.lightbox.set('tween',{duration: this.options.fadeSpeed});
 77 		//if fading fade out initially
 78 		if(this.options.fade)
 79 			this.lightbox.fade(0);
 80 		//lightbox close button
 81 		this.closeButton = $(lightbox+'Close');
 82 		this.closeButton.set('tween',{duration: this.options.fadeSpeed});
 83 		if(this.options.fade)
 84 			this.closeButton.fade(0);
 85 		//set up close butter to trigger
 86 		this.closeButton.addEvent('click',function() { this.trigger() }.bind(this));
 87 		//get lightbox content
 88 		this.content = $$('.'+lightbox+'Content');
 89 		
 90 	},
 91 	/**
 92 	  * @function used to show or hide the lightbox layer
 93 	  */
 94 	trigger: function() { 
 95 		if(this.hidden) {
 96 			//show lightbox and content blocks
 97 			this.lightbox.setStyle('display','block');
 98 			this.closeButton.setStyle('display','block');
 99 			this.content.setStyle('display','block');
100 			if(this.options.fade) {
101 				//fade in lightbox
102 				this.lightbox.fade(.5);
103 				this.closeButton.fade(1);
104 			}
105 			this.fireEvent('show');
106 			//set visability flag
107 			this.hidden = false;
108 		} else {
109 			this.fireEvent('hide');
110 		}
111 	},
112 	/**
113 	  * @function this will begin the lightbox fade
114 	  */
115 	fadeLightbox: function() {
116 		if(this.options.fade) {
117 			//fade out lightbox
118 			this.lightbox.fade(0);
119 			//fade closebutton with a delay
120 			(function() { this.closeButton.fade(0); }.bind(this)).delay(this.options.fadeSpeed);
121 		}
122 		this.removeLightbox.delay(this.options.removeDelay,this);
123 	},
124 	/**
125 	  * @function removeLightbox: removes lightbox from display
126 	  */
127 	removeLightbox: function() {
128 		//hide lightbox and content
129 		this.content.setStyle('display','none');
130 		this.closeButton.setStyle('display','none');
131 		this.lightbox.setStyle('display','none');
132 		//set visability flag
133 		this.hidden = true;
134 		this.fireEvent('remove');
135 	}
136 });

138 /**
139  * @class The ConfirmBox Class creates a LightBox object used to ask the user a simple question before continuing with processing
140  * @birth April 22nd, 2010
141  * @author Corbin Tarrant
142  *  ___            ___         __ 
143  *    |   /\  |\/| |       __   |_/  |__   -   _
144  *  _|_ /  \ |  | |___  |__|  | \  |__|  |  |  |
145  * 
146  * @link http://www.IAmCorbin.net
147  * @version 1.0
148  *
149  * @requires MooTools 1.2, LightBox
150  * {@link http://mootools.net/}
151  *
152  * @package MooKit
153  * 
154  *
155  * @property	{bool}		options.name 			The name of the confirmationBox
156  * @property	{bool}		options.font 			The Font
157  * @property	{bool}		options.fontSize		The Font size
158  * @property	{bool}		options.back 			The color of the lightbox
159  * @property	{bool}		options.color 			text color
160  * @property	{bool}		options.boxBorder 		border style of the box
161  * @property	{bool}		options.boxCurved 		Used to add an optional classname to apply a curved style to the box and buttons, defaults to an empty string
162  * @property	{bool}		options.boxBack 		background color of the box
163  * @property	{bool}		options.boxButtonColor	Box Button color
164  * @property	{bool}		options.boxMSG 		The Box Message
165  * @property	{bool}		options.boxYES 		The Confirm Button Text
166  * @property	{bool}		options.boxNO			The Cancel Button Text
167  * @property	{$empty}	options.onDisplay		Event fired when ConfirmBox is displayed
168  * @property	{$empty}	options.onConfirm		Event fired when Confirm Button is Clicked
169  */
170 var ConfirmBox = new Class({
171 	Implements: [Options,Events],
172 	options: {
173 		name: 'confirmBox',
174 		font: 'Monospace',
175 		fontSize: '23px',
176 		back: '#AAA',
177 		color: '#FFF',
178 		boxBorder: 'solid black 2px',
179 		boxCurved: '',
180 		boxBack: '#333',
181 		boxButtonColor: '#555',
182 		boxMSG: 'Are You Sure?',
183 		boxYES: 'YES',
184 		boxNO: 'NO '
185 		/*
186 		onDisplay: $empty,
187 		onConfirm: $empty
188 		*/
189 	},
190 	/**
191 	  * @constructor
192 	  *
193 	  * Creates the elements needs for the confirmation box using the passed in options, then creates a lightbox with the element and trggers it
194 	  *
195 	  * @param {string[]} 	options 		passed in options
196 	  */
197 	initialize: function(options) {
198 		this.setOptions(options);
199 		
200 		//The Background Layer
201 		new Element('div',{
202 			id: this.options.name,
203 			styles: { 
204 				position: 'fixed', left: '0', top: '0',
205 				width: '100%', height: '100%',
206 				display: 'none',
207 				background: this.options.back,
208 				zIndex: '10000'
209 			},
210 			
211 		}).inject(document.body);
212 		//Button Styles
213 		buttonStyles = new Hash({
214 			float: 'left',
215 			fontWeight: 'bold', fontSize: this.options.fontSize, fontFamily: this.options.font,
216 			cursor: 'pointer',
217 			border: 'solid black 1px',
218 			background: this.options.boxButtonColor,
219 			marginLeft: '30px',
220 			margin: '20px',
221 			padding: '10px',
222 			position: 'absolute', left: '10%', bottom: '0'
223 		});
224 		//The Confirm Button
225 		confirmButton = new Element('span',{
226 			id: this.options.name+'Confirm',
227 			styles: buttonStyles,
228 			class: this.options.boxCurved,
229 			html: this.options.boxYES
230 		});
231 		//The Close Button
232 		buttonStyles.erase('left');
233 		buttonStyles.set('right','10%');
234 		closeButton = new Element('span',{
235 			id: this.options.name+'Close',
236 			styles: buttonStyles,
237 			class: this.options.boxCurved,
238 			html: this.options.boxNO
239 		});
240 		//The Message Box
241 		confirmBox = new Element('div',{
242 			class: this.options.boxCurved+' '+this.options.name+'Content',
243 			styles: {
244 				position: 'fixed', left: '40%', top: '100px',
245 				width: this.options.boxMSG.length*this.options.fontSize+'px', height: confirmButton.getStyle('lineHeight').toInt()*5+'px',
246 				fontFamily: this.options.font, fontSize: this.options.fontSize, color: this.options.color,
247 				display: 'none',
248 				border: this.options.boxBorder,
249 				backgroundColor: this.options.boxBack,
250 				padding: '10px',
251 				zIndex: '10001'
252 			},
253 			html: '<div>'+this.options.boxMSG+'</div><br />'
254 		});
255 		//inject buttons into box
256 		confirmButton.inject(confirmBox);
257 		closeButton.inject(confirmBox);
258 		//add confirmBox to DOM
259 		confirmBox.inject(document.body);
260 		//Create the Lightbox
261 		this.confirmLB = new LightBox(this.options.name, {
262 			onShow: function() {
263 				this.fireEvent('display');
264 			}.bind(this),
265 			onHide: function() { 
266 				this.fadeLightbox.delay('200',this);
267 			},
268 			onRemove: function() {
269 				//destroy confirmation box
270 				if($(this.options.name))	$(this.options.name).destroy();
271 				if($$('.'+this.options.name+'Content')) $$('.'+this.options.name+'Content').destroy();
272 			}.bind(this),	
273 		});
274 		this.confirmLB.trigger();
275 		
276 		//Add Confirm Button Event
277 		$(this.options.name+'Confirm').addEvent('click', function() {
278 			this.fireEvent('confirm');
279 			this.confirmLB.trigger();
280 		}.bind(this));
281 		
282 	}
283 });
