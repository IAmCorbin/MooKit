1 /**
  2   * @function update the application on user status change (log in/out)
  3   */
  4 function updateApp() {
  5 	refreshMenu();
  6 	refreshUserInfo();
  7 }
  8 /** @function refresh the site menu */ 
  9 function refreshMenu() {
 10 	new Request.HTML({
 11 		url: 'codeSite/php/updateMenu.php',
 12 		onSuccess: function(responseTree, responseElements, responseHTML, responseJavaScript) { 
 13 			$('mainNav').fade(0);
 14 			(function() { $('mainNav').set('html',responseHTML); }).delay(500);
 15 			(function() { $('mainNav').fade(1); }).delay(500);
 16 			//update menu javascript -- delay for a second to make sure the menu html has finished setting
 17 			(function() { addAssets([""],["codeSite/js/menu.js"]); } ).delay(1000);
 18 		}
 19 	}).send();
 20 }
 21 /** @function refresh the user info box */ 
 22 function refreshUserInfo() {
 23 	new Request({
 24 			url: 'codeSite/php/updateUserInfo.php',
 25 			onSuccess: function(response) { 
 26 				userInfo = $('userInfo');
 27 				userInfo.tween('opacity','0');
 28 				(function() { userInfo.set('html',response); }).delay(500);
 29 				(function() { userInfo.tween('opacity','1'); }).delay(1000);
 30 			}
 31 	}).send();
 32 }
